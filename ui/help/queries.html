Example Queries


    -- last 10 metrics published for each dimension within last 5hrs
    select * from /.*/ where time > now() - 5h limit 10

    -- last 10 metrics reported from specific source over last 1h
    select * from /^demo.*/ where time > now() - 1h limit 10

    -- CPU metrics published over last 1h in down-sampled to 5min 
    select min(value) as MinVal,
           PERCENTILE(value, 25) as LowPercentile,
           mean(value) as MedVal,
           PERCENTILE(value, 75) as HighPercentile,
           max(value) as MaxVal
    from /^demo.*/
    where time > now() - 1h
    group by time(5m)
